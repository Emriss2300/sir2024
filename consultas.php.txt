<form id="formularioEtiquetas">
    <label for="codigo">Código del Producto:</label>
    <input type="text" id="codigo" required>
    
    <label for="cantidad">Cantidad de Etiquetas:</label>
    <input type="number" id="cantidad" min="1" value="1" required>
    
    <input type="submit" value="Agregar Etiquetas">
</form>

<div class="etiquetas-container" id="etiquetasContainer"></div>

<button class="no-print" onclick="imprimirEtiquetas()">Imprimir Etiquetas</button>

<script>
    document.getElementById('formularioEtiquetas').addEventListener('submit', function(event) {
        event.preventDefault();
        
        const codigo = document.getElementById('codigo').value;
        const cantidad = parseInt(document.getElementById('cantidad').value);
        const container = document.getElementById('etiquetasContainer');

        for (let i = 0; i < cantidad; i++) {
            const etiqueta = document.createElement('div');
            etiqueta.className = 'etiqueta';
            etiqueta.innerHTML = `
                <h2>${codigo}</h2>
                <p>Descripción del Producto</p>
                <p>Cantidad: ${cantidad}</p>
                <p>Precio Uni.: $XX.XX</p>
                <p class="codigo-barra" style="font-family: 'Bar-Code 39'; font-size: 32px;">*${codigo}*</p>
            `;
            container.appendChild(etiqueta);
        }

        // Limpiar el formulario
        this.reset();
    });

    function imprimirEtiquetas() {
        window.print();
    }
</script>
